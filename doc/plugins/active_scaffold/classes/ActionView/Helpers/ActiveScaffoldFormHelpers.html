13:     def in_subform?(column, parent_record)
14:       return true unless column.association
15: 
16:       # Polymorphic associations can't appear because they *might* be the reverse association, and because you generally don't assign an association from the polymorphic side ... I think.

17:       return false if column.polymorphic_association?
18: 
19:       # We don't have the UI to currently handle habtm in subforms

20:       return false if column.association.macro == :has_and_belongs_to_many
21: 
22:       # A column shouldn't be in the subform if it's the reverse association to the parent

23:       return false if column.association.reverse_for?(parent_record.class)

25: 
26:       return true
27:     end
